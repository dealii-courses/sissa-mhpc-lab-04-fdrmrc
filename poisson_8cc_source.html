101  dim == 2 ? "x,y" :

120 template <int dim>
121 void
122 Poisson<dim>::assemble_system()
123 {
124  QGauss<dim> quadrature_formula(fe->degree + 1);
125  FEValues<dim> fe_values(*fe,
126  quadrature_formula,
127  update_values | update_gradients |
128  update_quadrature_points | update_JxW_values);
129  const unsigned int dofs_per_cell = fe->n_dofs_per_cell();
130  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
131  Vector<double> cell_rhs(dofs_per_cell);
132  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
133  for (const auto &cell : dof_handler.active_cell_iterators())
134  {
135  fe_values.reinit(cell);
136  cell_matrix = 0;
137  cell_rhs = 0;
138  for (const unsigned int q_index : fe_values.quadrature_point_indices())
139  {
140  const double current_coefficient = stiff_coefficient.value(fe_values.quadrature_point(q_index));
141  // std::cout << current_coefficient << "\n";
142  for (const unsigned int i : fe_values.dof_indices())
143  for (const unsigned int j : fe_values.dof_indices())
144  cell_matrix(i, j) +=
145  (current_coefficient * // a(x_q)
146  fe_values.shape_grad(i, q_index) * // grad phi_i(x_q)
147  fe_values.shape_grad(j, q_index) * // grad phi_j(x_q)
148  fe_values.JxW(q_index)); // dx
149  for (const unsigned int i : fe_values.dof_indices())
150  cell_rhs(i) += (fe_values.shape_value(i, q_index) * // phi_i(x_q)
151  forcing_term.value(
152  fe_values.quadrature_point(q_index)) * // f(x_q)
153  fe_values.JxW(q_index)); // dx
154  }
155  cell->get_dof_indices(local_dof_indices);
156  for (const unsigned int i : fe_values.dof_indices())
157  for (const unsigned int j : fe_values.dof_indices())
158  system_matrix.add(local_dof_indices[i],
159  local_dof_indices[j],
160  cell_matrix(i, j));
161  for (const unsigned int i : fe_values.dof_indices())
162  system_rhs(local_dof_indices[i]) += cell_rhs(i);
163  }
164  std::map<types::global_dof_index, double> boundary_values;
165  VectorTools::interpolate_boundary_values(dof_handler,
166  0,
167  boundary_condition,
168  boundary_values);
169  MatrixTools::apply_boundary_values(boundary_values,
170  system_matrix,
171  solution,
172  system_rhs);
173 }

201 template <int dim>
202 void
203 Poisson<dim>::run()
204 {
205  make_grid();
206 
207  for (unsigned int cycle = 0; cycle < n_refinement_cycles; ++cycle)
208  {
209  setup_system();
210  assemble_system();
211  solve();
212  compute_error();
213  output_results(cycle);
214  if (cycle < n_refinement_cycles - 1)
215  { // avoid refine the grid and use it afterwards
216  refine_grid();
217  }
218  }
219  error_table.output_table(std::cout);
220 }
